gulp API

gulp.src

说明

gulp.src将glob或glob数组匹配的文件作为VinyI files流返回。

语法

gulp.src(globs[, options])

示例

gulp.src('client/templates/*.jade')
  .pipe(jade())
  .pipe(minify())
  .pipe(gulp.dest('build/minified_templates'));

gulp.dest

说明

gulp.dest将文件流输出到目标路径。

语法

gulp.dest(path[, options])

示例

gulp.src('client/templates/*.jade')
  .pipe(jade())
  .pipe(minify())
  .pipe(gulp.dest('build/minified_templates'));

gulp.task

说明

gulp.task定义了一个任务。

语法

gulp.task(name[, deps][, fn])

示例

gulp.task('pack', ['build', 'merge', 'minify'], function() {
});

gulp-babel

  • 安装
    npm install --save-dev gulp
  • 示例
    var gulp = require('gulp');
var babel = require("gulp-babel");

gulp.task("default", function() {
  return gulp.src("src/app.js")
    .pipe(babel({
      presets: ["es2015", "react"]
    }))
    .pipe(gulp.dest("dist"));
});

gulp-livereload

  • 安装
    npm install --save-dev gulp-livereload
  • 示例
    var gulp = require("gulp");
var less = require("gul-less");
var livereload = require("gulp-livereload");

gulp.task("less", function() {
  gulp.src("less/*.less")
    .pipe(less())
    .pipe(gulp.dest("css"))
    .pipe(livereload());
});

gulp.task("watch", function() {
  livereload.listen();
  gulp.watch("less/*.less", ["less"]);
});

gulp-webserver

  • 安装
    npm install --save-dev gulp-webserver
  • 示例
    var gulp = require("gulp");
var babel = require("gulp-babel");
var webpack = require("gulp-webpack");
var webserver = require("gulp-webserver");
var livereload = require("gulp-livereload");

gulp.task("default", ["build-js", "build-html"], function() {
});

gulp.task("build-js", function() {
        return gulp.src("src/*.js")
                .pipe(babel({
                        presets: ["es2015", "react", "stage-0"]
                }))
                .pipe(gulp.dest("build"))
                .pipe(webpack({
                        module:  {
                                loaders: [{
                                        test: /\.js$/,
                                        exclude: /node_modules/,
                                        loader: "babel-loader",
                                }],
                        },
                        output: {
                                filename: "bundle.js",
                        },
                }))
                .pipe(gulp.dest("build"));
});

gulp.task("build-html", function() {
        return gulp.src("src/*.html")
                .pipe(gulp.dest("build"));
});


gulp.task("init-dev-env", ["start-server", "watch-file-change"]);

gulp.task("watch-file-change", function() {
        gulp.watch("src/*.js", ["build-js"])
        gulp.watch("src/*.html", ["build-html"])
});

gulp.task("start-server", function() {
        gulp.src("build")
                .pipe(webserver({
                        livereload: true,
                        open: true,
                        host: "0.0.0.0",
                }));
});
