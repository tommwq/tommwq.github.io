Room简介

Room是jetpack中的一个组件，用于在Android软件中支持ORM功能。Room包含3个主要的组件：

  • Database。表示一个数据库。
  • Entity。数据表的结构。
  • DAO。提供访问数据库的接口。

在使用Room时，这三个组件都会用到。

Entity

首先来看Entity，Entity定义了数据表结构，下面是一个例子： 

@Entity(tableName="users")
data class User (
    @PrimaryKey var uid: Int,
    @ColumnInfo(name = "first_name") var firstName: String?,
    @ColumnInfo(name = "last_name") var lastName: String?
    @Ignore var picture: Bitmap?
)

现在我们来看一个更复杂的例子： 

@Entity(tableName = "plants")
data class Plant(
    @PrimaryKey @ColumnInfo(name = "id") val plantId: String,
    val name: String,
    val description: String,
    val growZoneNumber: Int,
    val wateringInterval: Int = 7,
    val imageUrl: String = ""
) {
    fun shouldBeWatered(since: Calendar, lastWateringDate: Calendar) =
        since > lastWateringDate.apply { add(DAY_OF_YEAR, wateringInterval) }

    override fun toString() = name
}

从这个例子里可以看到，Entity的成员可以有默认值，Entity也可以拥有行为。

Dao

Entity定义了记录结构，而要访问记录，则需要使用Dao： 

@Dao
interface UserDao {
    @Query("SELECT * FROM user")
    fun getAll(): List<User>

    @Query("SELECT * FROM user WHERE uid IN (:userIds)")
    fun loadAllByIds(userIds: IntArray): List<User>

    @Query("SELECT * FROM user WHERE first_name LIKE :first AND last_name LIKE :last LIMIT 1")
    fun findByName(first: String, last: String): User

    @Insert
    fun insertAll(vararg users: User)

    @Delete
    fun delete(user: User)
}

@Dao
interface PlantDao {
    @Query("SELECT * FROM plants ORDER BY name")
    fun getPlants(): LiveData<List<Plant>>

    @Query("SELECT * FROM plants WHERE growZoneNumber = :growZoneNumber ORDER BY name")
    fun getPlantsWithGrowZoneNumber(growZoneNumber: Int): LiveData<List<Plant>>

    @Query("SELECT * FROM plants WHERE id = :plantId")
    fun getPlant(plantId: String): LiveData<Plant>

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    fun insertAll(plants: List<Plant>)
}

我们看到Dao是一个接口，要如何获得实现类的实例呢？方法是使用Database。Database可以返回Dao，让我们操作Entity： 

@Database(entities = [User::class], version = 1)
abstract class AppDatabase: RoomDatabase() {
    abstract fun userDao(): UserDao
}

而Database可以通过Room.databaseBuilder得到： 

val db = Room.databaseBuilder(
    applicationContext,
    AppDatabase::class.java,
    "database-name"
).build()

Room类在包androidx.room中。

一个@Database类需要满足：

  • 是一个派生自RoomDatabase的抽象类。
  • 包含一个没有参数的抽象方法，返回一个@DAO对象。

示例： 

apply plugin: 'kotlin-kapt'

dependencies {
    def room_version = "2.1.0-alpha04"
    implementation "androidx.room:room-runtime:$room_version"
    kapt "android.arch.persistence.room:compiler:1.0.0"
}

@Dao
abstrace class Foo: RoomDatabase() {
    @Transaction
    fun insertAndDelete() { ... }
    @Update
    fun updateUsers(vararg users: User)
    @Insert
    fun foo(user: User, friends: List<User>)
}

@Insert(onConflict=OnConflictStrategy.REPLACE)
